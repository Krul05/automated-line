Рабочий контроллер линии Montrac.

Контроллер хранит логическую карту станций и перегонов, принимает IRM-кадры присутствия от serial-воркеров, отправляет команды START FORWARD, применяет автоматические режимы и сохраняет редактируемую конфигурацию станций/режимов.

 95class SegmentState:
 96    """Текущее состояние перегона между двумя соседними станциями."""

 98    from_station: int
 99    to_station: int
100    occupied_by: int | None = None

105    def shuttle_count(self) -> int:
106        """Вернуть количество тележек, которые сейчас числятся в перегоне."""
107        self._sync_from_legacy()
108        return len(self.shuttle_ids)
109
110    def add_shuttle(self, shuttle_id: int, since: float) -> None:
111        """Отметить тележку как движущуюся по этому перегону."""
112        self._sync_from_legacy()
113        if shuttle_id in self.shuttle_ids:
114            self.shuttle_since[shuttle_id] = since
115        else:
116            self.shuttle_ids.append(shuttle_id)
117            self.shuttle_since[shuttle_id] = since
118        self._sync_legacy()
119
120    def remove_shuttle(self, shuttle_id: int) -> bool:
121        """Удалить тележку из перегона и вернуть, была ли она там."""
122        self._sync_from_legacy()
123        if shuttle_id not in self.shuttle_ids:
124            return False
125        self.shuttle_ids = [item for item in self.shuttle_ids if item != shuttle_id]
126        self.shuttle_since.pop(shuttle_id, None)
127        self._sync_legacy()
128        return True
129
130    def _sync_from_legacy(self) -> None:
131        if self.occupied_by is None or self.occupied_by in self.shuttle_ids:
132            return
133        self.shuttle_ids.insert(0, self.occupied_by)
134        if self.since is not None:
135            self.shuttle_since.setdefault(self.occupied_by, self.since)
136
137    def _sync_legacy(self) -> None:
138        if not self.shuttle_ids:
139            self.occupied_by = None
140            self.since = None
141            return
142        self.occupied_by = self.shuttle_ids[0]
143        known_times = [self.shuttle_since[item] for item in self.shuttle_ids if item in self.shuttle_since]
144        self.since = min(known_times) if known_times else None

156class SerialWorker(threading.Thread):
157    """Фоновый поток чтения/записи для COM-порта одной станции."""
158
159    def __init__(self, controller: "MontracController", station: StationConfig):
160        super().__init__(name=f"station-{station.index}-reader", daemon=True)
161        self.controller = controller
162        self.station = station
163        self._stop_event = threading.Event()
164        self._write_lock = threading.Lock()
165        self._serial: Any = None
166        self._buffer = bytearray()
167        self.sent_frames: list[bytes] = []
168
169    def run(self) -> None:
170        """Открывать порт станции, читать IRM-кадры и переподключаться при сбоях."""
171        if self.station.mock:
172            self.controller.set_connection(self.station.index, True, "mock port")
173            while not self._stop_event.wait(0.5):
174                pass
175            return
176
177        while not self._stop_event.is_set():
178            if self._serial is None:
179                self._open_port()
180                if self._serial is None:
181                    self._stop_event.wait(self.station.reconnect_interval)
182                    continue
183
184            try:
185                waiting = getattr(self._serial, "in_waiting", 0)
186                if waiting:
187                    data = self._serial.read(waiting)
188                    self._buffer.extend(data)
189                    for message in extract_messages(self._buffer):
190                        self.controller.on_message(self.station.index, message)
191                else:
192                    self._stop_event.wait(self.station.read_timeout)
193            except Exception as exc:  # pragma: no cover - hardware path
194                self.controller.set_connection(self.station.index, False, str(exc))
195                self._close_port()
196                self._stop_event.wait(self.station.reconnect_interval)
197
198    def _open_port(self) -> None:
199        try:
200            import serial  # type: ignore
201
202            self._serial = serial.Serial(
203                self.station.port,
204                self.station.baudrate,
205                timeout=self.station.read_timeout,
206            )
207            self._serial.reset_input_buffer()
208            self.controller.set_connection(self.station.index, True, None)
209        except Exception as exc:  # pragma: no cover - hardware path
210            self._serial = None
211            self.controller.set_connection(self.station.index, False, str(exc))
212
213    def send(self, frame: bytes) -> None:
214        """Записать сырой IRM-кадр в порт станции или сохранить его в mock-режиме."""
215        with self._write_lock:
216            if self.station.mock:
217                self.sent_frames.append(frame)
218                return
219            if self._serial is None:
220                raise RuntimeError(f"{self.station.port} is not connected")
221            self._serial.write(frame)
222            self._serial.flush()
223
224    def stop(self) -> None:
225        """Передать воркеру сигнал остановки и закрыть serial-порт."""
226        self._stop_event.set()
227        self._close_port()
228
229    def _close_port(self) -> None:
230        with self._write_lock:
231            if self._serial is not None:
232                try:
233                    self._serial.close()
234                finally:
235                    self._serial = None

238class MontracController:
239    """Координировать состояние станций, проверки движения, режимы и сохранение.
240
241    Тележка может быть отправлена только когда следующий перегон не содержит
242    тележек. Это правило одинаково применяется к кнопкам станций, автоматическим
243    режимам и API-вызовам.
244    """
245
246    def __init__(self, config: LineConfig, config_path: Path | None = None, force_mock: bool = False):
247        self.config = config
248        self.config_path = config_path
249        self.force_mock = force_mock
250        self.active_mode_id: str | None = None
251        self._lock = threading.RLock()
252        self._stop = threading.Event()
253        self._scheduler: threading.Thread | None = None
254        self._started = False
255        self._station_order: list[int] = []
256        self._station_config: dict[int, StationConfig] = {}
257        self.stations: dict[int, StationState] = {}
258        self.segments: dict[tuple[int, int], SegmentState] = {}
259        self.workers: dict[int, SerialWorker] = {}
260        self._mode_config: dict[str, ModeConfig] = {}
261        self.events: list[Event] = []
262        self._rebuild_runtime_locked(config.stations)
263        self._rebuild_modes_locked(config.modes)
264
265    def start(self) -> None:
266        """Запустить все рабочие потоки станций и планировщик режима."""
267        with self._lock:
268            self._started = True
269            self._log("info", "Montrac controller started")
270            workers = list(self.workers.values())
271        for worker in workers:
272            worker.start()
273        self._scheduler = threading.Thread(target=self._scheduler_loop, name="montrac-scheduler", daemon=True)
274        self._scheduler.start()
275
276    def stop(self) -> None:
277        """Остановить планировщик и рабочие потоки станций с коротким ожиданием."""
278        self._stop.set()
279        self._started = False
280        workers = list(self.workers.values())
281        for worker in workers:
282            worker.stop()
283        for worker in workers:
284            if worker.is_alive():
285                worker.join(timeout=1)
286        if self._scheduler and self._scheduler.is_alive():
287            self._scheduler.join(timeout=1)
288        self._log("info", "Montrac controller stopped")
289
290    def set_connection(self, station_index: int, connected: bool, error: str | None) -> None:
291        """Обновить видимое состояние подключения для потока одной станции."""
292        with self._lock:
293            if station_index not in self.stations:
294                return
295            state = self.stations[station_index]
296            was_connected = state.connected
297            state.connected = connected
298            state.last_error = error
299            if connected and not was_connected:
300                self._log("info", f"{state.name} connected")
301            elif not connected and was_connected:
302                self._log("error", f"{state.name} disconnected: {error}")
303            elif not connected and error:
304                state.last_error = error
305
306    def set_mode(self, mode: str | OperatingMode | None) -> dict[str, Any]:
307        """Выбрать автоматический режим или вернуться в режим ожидания."""
308        with self._lock:
309            mode_id = mode.value if isinstance(mode, OperatingMode) else mode
310            if mode_id in (None, "", OperatingMode.IDLE.value):
311                self.active_mode_id = None
312                self._log("info", "Mode changed to idle")
313                return self.snapshot_locked()
314            if mode_id not in self._mode_config:
315                raise ValueError(f"Mode does not exist: {mode_id}")
316            self.active_mode_id = mode_id
317            self._log("info", f"Mode changed to {self._mode_config[mode_id].name}")
318            self._auto_release_due_locked(time.time())
319            return self.snapshot_locked()

337    def update_station_config(self, raw_stations: list[dict[str, Any]]) -> dict[str, Any]:
338        """Заменить и сохранить конфигурацию станций, когда линия пустая."""
339        with self._lock:
340            if self._line_has_active_shuttles_locked():
341                raise ValueError("Нельзя менять конфигурацию, пока есть занятые станции или перегоны")
342            new_stations = self._normalize_station_config_locked(raw_stations)
343            old_workers = list(self.workers.values())
344
345        for worker in old_workers:
346            worker.stop()
347        for worker in old_workers:
348            if worker.is_alive():
349                worker.join(timeout=1)
350
351        with self._lock:
352            self.config.stations = new_stations
353            self.config.modes = normalize_modes_for_stations(self.config.modes, new_stations, self.config.hold_seconds)
354            if self.config_path is not None:
355                self.config_path = save_config(self.config, self.config_path)
356            self._rebuild_runtime_locked(new_stations)
357            self._rebuild_modes_locked(self.config.modes)
358            if self._started and not self._stop.is_set():
359                for worker in self.workers.values():
360                    worker.start()
361            self._log("info", "Station configuration updated")
362            state = self.snapshot_locked()
363            state["saved"] = True
364            return state

366    def update_modes_config(self, raw_modes: list[dict[str, Any]]) -> dict[str, Any]:
367        """Заменить конфигурацию режимов, нормализовать задержки и сохранить."""
368        with self._lock:
369            modes = normalize_modes_for_stations(parse_modes(raw_modes, self.config.stations), self.config.stations, self.config.hold_seconds)
370            self.config.modes = modes
371            self._rebuild_modes_locked(modes)
372            if self.active_mode_id not in self._mode_config:
373                self.active_mode_id = None
374            if self.config_path is not None:
375                self.config_path = save_config(self.config, self.config_path)
376            self._log("info", "Mode configuration updated")
377            state = self.snapshot_locked()
378            state["saved"] = True
379            return state

381    def on_message(self, station_index: int, message: IrmMessage) -> None:
382        """Обработать разобранное IRM-сообщение от потока станции."""
383        with self._lock:
384            if station_index not in self.stations:
385                return
386            station = self.stations[station_index]
387            station.message_count += 1
388            station.last_seen = time.time()
389            station.last_raw = message.hex
390            station.group = message.group
391
392            if message.command == COMMAND_PRESENCE:
393                self._handle_presence_locked(station_index, message)
394            else:
395                self._log("debug", f"{station.name} received command {message.command} for shuttle {message.shuttle_id}")

397    def inject_presence(self, station_index: int, shuttle_id: int, group: int = 1) -> dict[str, Any]:
398        """Подставить синтетическое сообщение присутствия для проверок без железа."""
399        frame = build_frame(group, shuttle_id, COMMAND_PRESENCE)
400        self.on_message(station_index, IrmMessage(group=group, shuttle_id=shuttle_id, command=COMMAND_PRESENCE, raw=frame))
401        with self._lock:
402            return self.snapshot_locked()

404    def _handle_presence_locked(self, station_index: int, message: IrmMessage) -> None:
405        """Применить кадр присутствия к состоянию станции и перегонов."""
406        now = time.time()
407        station = self.stations[station_index]
408        shuttle_id = message.shuttle_id
409
410        if (
411            station.released_shuttle_id == shuttle_id
412            and station.release_sent_at is not None
413            and now - station.release_sent_at < self.config.departure_grace_seconds
414        ):
415            return
416
417        for segment in self.segments.values():
418            if segment.remove_shuttle(shuttle_id):
419                self._log("info", f"Shuttle {shuttle_id} arrived at {station.name}")
420
421        for other in self.stations.values():
422            if other.index != station_index and other.occupied and other.shuttle_id == shuttle_id:
423                other.occupied = False
424                other.shuttle_id = None
425                other.waiting_since = None
426
427        if not station.occupied or station.shuttle_id != shuttle_id:
428            if station.occupied and station.shuttle_id != shuttle_id:
429                self._log(
430                    "warning",
431                    f"{station.name} changed occupied shuttle from {station.shuttle_id} to {shuttle_id}",
432                )
433            station.occupied = True
434            station.shuttle_id = shuttle_id
435            station.waiting_since = now
436            station.release_sent_at = None
437            station.released_shuttle_id = None
438            self._log("info", f"Shuttle {shuttle_id} detected at {station.name}")
439
440    def _scheduler_loop(self) -> None:
441        while not self._stop.wait(self.config.scheduler_interval_seconds):
442            with self._lock:
443                self._auto_release_due_locked(time.time())

445    def _auto_release_due_locked(self, now: float) -> None:
446        """Отправить тележки, у которых истекла задержка активного режима."""
447        if self.active_mode_id is None:
448            return
449        mode = self._mode_config.get(self.active_mode_id)
450        if mode is None:
451            self.active_mode_id = None
452            return
453
454        for station_index in self._station_order:
455            station = self.stations[station_index]
456            if not station.occupied or station.waiting_since is None:
457                continue
458            hold_seconds = mode.station_delays.get(station_index, 0.0)
459            if now - station.waiting_since >= hold_seconds:
460                self._release_station_locked(station_index, f"mode {mode.name}")
461
462    def _release_next_locked(self, reason: str) -> dict[str, Any]:
463        occupied = [
464            station
465            for station in self.stations.values()
466            if station.occupied and station.shuttle_id is not None and station.waiting_since is not None
467        ]
468        occupied.sort(key=lambda item: (item.waiting_since or 0, item.index))
469
470        blockers: list[dict[str, Any]] = []
471        for station in occupied:
472            ok, blocker = self._can_release_locked(station.index)
473            if ok:
474                return self._release_station_locked(station.index, reason)
475            blockers.append({"station": station.name, "reason": blocker})
476
477        message = "No shuttle can be released safely"
478        self._log("warning", message)
479        return {"released": False, "message": message, "blockers": blockers}
480
481    def _release_station_locked(self, station_index: int, reason: str) -> dict[str, Any]:
482        """Отправить START FORWARD и перенести тележку из станции в перегон."""
483        ok, blocker = self._can_release_locked(station_index)
484        station = self.stations.get(station_index)
485        if station is None:
486            return {
487                "released": False,
488                "station": station_index,
489                "reason": "station does not exist",
490            }
491        if not ok:
492            return {
493                "released": False,
494                "station": station.name,
495                "reason": blocker,
496            }
497
498        target = self.next_station(station_index)
499        assert target is not None
500        segment = self.segments[(station_index, target)]
501        shuttle_id = station.shuttle_id
502        assert shuttle_id is not None
503        frame = build_frame(station.group, shuttle_id, COMMAND_START_FORWARD)
504
505        try:
506            self.workers[station_index].send(frame)
507        except Exception as exc:
508            station.last_error = str(exc)
509            self._log("error", f"Could not send START FORWARD to {station.name}: {exc}")
510            return {
511                "released": False,
512                "station": station.name,
513                "shuttleId": shuttle_id,
514                "reason": str(exc),
515            }
516
517        now = time.time()
518        station.occupied = False
519        station.shuttle_id = None
520        station.waiting_since = None
521        station.release_sent_at = now
522        station.released_shuttle_id = shuttle_id
523        station.last_command_hex = frame.hex().upper()
524        for other_segment in self.segments.values():
525            if other_segment is not segment:
526                other_segment.remove_shuttle(shuttle_id)
527        segment.add_shuttle(shuttle_id, now)

751def parse_modes(raw_modes: Any, stations: list[StationConfig]) -> list[ModeConfig]:
752    """Разобрать объекты режимов из JSON-конфига или тела API-запроса."""
753    if raw_modes is None:
754        return []
755    if not isinstance(raw_modes, list):
756        raise ValueError("modes must be a list")
757
758    modes: list[ModeConfig] = []
759    used_ids: set[str] = set()
760    for position, raw in enumerate(raw_modes, start=1):
761        if not isinstance(raw, dict):
762            raise ValueError("mode must be an object")
763        name = str(raw.get("name") or f"Режим {position}").strip()
764        mode_id = str(raw.get("id") or slugify_mode_id(name, position)).strip()
765        if not mode_id:
766            mode_id = f"mode_{position}"
767        if mode_id in used_ids:
768            raise ValueError(f"Mode id repeats: {mode_id}")
769        used_ids.add(mode_id)
770
771        raw_delays = raw.get("stationDelays", raw.get("delays", {}))
772        delays = parse_station_delays(raw_delays)
773        modes.append(ModeConfig(id=mode_id, name=name, station_delays=delays))
774
775    return modes

778def parse_station_delays(raw_delays: Any) -> dict[int, float]:
779    """Разобрать задержки станций из dict- или list-формата."""
780    delays: dict[int, float] = {}
781    if isinstance(raw_delays, dict):
782        iterator = raw_delays.items()
783    elif isinstance(raw_delays, list):
784        iterator = []
785        for item in raw_delays:
786            if isinstance(item, dict):
787                iterator.append((item.get("stationIndex"), item.get("seconds", 0)))
788    else:
789        iterator = []
790
791    for raw_index, raw_seconds in iterator:
792        if raw_index in (None, ""):
793            continue
794        index = int(raw_index)
795        seconds = float(raw_seconds or 0)
796        if seconds < 0:
797            raise ValueError("Delay seconds must be non-negative")
798        delays[index] = seconds
799    return delays

831    """Создать стабильный ASCII-идентификатор режима из читаемого названия."""

837            result.append("_")
